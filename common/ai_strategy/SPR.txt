# make italy and germany want to send stuff to the fascist side of SCW
ITA_and_GER_help_SPA_win_civil_war = {
	reversed = yes
	enable = {
		date < 1939.1.1
		country_exists = SPR
		SPR = {
			has_government = fascism
		}
		has_government = fascism
		SPR = { has_civil_war = yes }
		NOT = { has_dlc = "La Resistance" }
		NOT = { has_global_flag = scw_over }
	}
	abort = {
		OR = {
			date > 1939.1.1
			NOT = { has_government = fascism }
			SPR = { has_civil_war = no }
			has_global_flag = scw_over
		}
	}
	
	ai_strategy = {
		type = send_volunteers_desire
		id = SPR
		value = 50
	}

}

	}
	
	enable = {
		OR = {
			has_global_flag = SPR_civil_war_startup # Don't attack for 30 days
			#AND = {
			#	SPR = { controls_state = 41 controls_state = 165 }
			#	JAP = { has_war = no }
			#}
			has_idea = SPR_death_before_surrender # Don't attack if you're Republican Spain before 39

		}

	ai_strategy = {
		type = front_control

		country_trigger = {
			original_tag = SPR
		}
		
		execute_order = no
		manual_attack = no
	}
}

# Prevent Nationalist Spain to join any faction on historical mode
nationalist_spain_allies_nobody_on_historical = {
	enable = {
		has_global_flag = scw_over
		has_dlc = "La Resistance"
		tag = SPA
		OR = {
			AND = {
				is_historical_focus_on = yes
				has_game_rule = {
					rule = SPR_ai_behavior 
					option = DEFAULT
				}
			}
			has_country_flag = SPR_AI_RANDOM_NATIONALIST_HISTORICAL
			has_game_rule = {
				rule = SPR_ai_behavior 
				option = NATIONALIST_HISTORICAL
			}
		}
	}
	abort = {
		OR = {
			SPA = { has_war = yes }
			SPA = { is_in_faction = yes }
			GER = { NOT = { has_government = fascism } }
			ENG = { NOT = { has_government = democratic } }
		}
	}
	
	ai_strategy = {
		type = alliance
		id = GER
		value = -200
	}

	ai_strategy = {
		type = alliance
		id = ENG
		value = -200
	}
}
