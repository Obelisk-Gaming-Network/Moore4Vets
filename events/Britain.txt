		name = britain.14.a

			add_opinion_modifier = { target = ENG modifier = western_betrayal }
		}		
		if = {

				country_exists = FRA
			}

		}

			limit = {
				has_guaranteed = YUG
			}

   				active = no
  			}

				add_opinion_modifier = { target = FRA modifier = western_betrayal }
				country_event = { id = yugoslavia.5 hours = 6 }
			}			
		}
	}
}
